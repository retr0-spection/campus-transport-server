Bus Schedules API Documentation

/api/bus-schedules

{
  "routeId": string (optional),
  "stopId": string (optional),
  "date": string (YYYY-MM-DD format, optional, defaults to current date),
  "startTime": string (HH:MM format, optional),
  "endTime": string (HH:MM format, optional)
}

Request Parameters

{
  "status": string,
  "data": {
    "schedules": [
      {
        "scheduleId": string,
        "routeId": string,
        "routeName": string,
        "stops": [
          {
            "stopId": string,
            "stopName": string,
            "arrivalTime": string (HH:MM format),
            "departureTime": string (HH:MM format),
            "coordinates": {
              "latitude": number,
              "longitude": number
            }
          }
        ],
        "busInfo": {
          "busId": string,
          "capacity": number,
          "accessibilityFeatures": [string],
          "currentLocation": {
            "latitude": number,
            "longitude": number
          },
          "estimatedArrival": string (HH:MM format)
        }
      }
    ],
    "pagination": {
      "currentPage": number,
      "totalPages": number,
      "totalItems": number,
      "itemsPerPage": number
    }
  },
  "message": string
}

Example Request:

GET /api/bus-schedules?routeId=R001&date=2024-08-21&startTime=08:00&endTime=10:00
